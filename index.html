<script>

// GAME STATE

let currentEvent = 0;

let health = 10;

let gold = 10;

let progress = 0;


// EVENTS ARRAY

const events = [

  {

    title: "1. The Letter That Changes Everything",

    text: "It arrives folded, dusty, and urgent. “Gold has been discovered in California…” Within days, your town is buzzing. Shops close. Neighbors vanish overnight. One man sells his farm for almost nothing just to leave faster.",

    image: "images/letter.png",

    choices: [

      {text: "Join a wagon train (overland)", flavor:"Lose 2 health, Gain 10 gold, Progress +1", effects:()=>{health-=2; gold+=10; progress+=1;}},

      {text: "Sail around South America", flavor:"Lose 1 health, Spend 20 gold, Progress +2", effects:()=>{health-=1; gold-=20; progress+=2;}},

      {text: "Cross through Panama", flavor:"Lose 3 health, Spend 10 gold, Progress +2", effects:()=>{health-=3; gold-=10; progress+=2;}}

    ]

  },

  {

    title: "2. Packing for the Unknown",

    text: "You stare at your limited supplies. You can only afford ONE more item. Whatever you choose…you’ll live with.",

    image: "images/packing.png",

    choices: [

      {text:"Extra food supplies", flavor:"Health +2 (applies to next health loss)", effects:()=>{health+=2;}},

      {text:"High-quality mining tools", flavor:"Gain 15 gold (first mining success only)", effects:()=>{gold+=15;}},

      {text:"A sturdy wagon upgrade", flavor:"Progress +2 (applies to next travel event)", effects:()=>{progress+=2;}}

    ]

  },

  {

    title: "3. The River Crossing",

    text:"The current is stronger than it looks. Halfway across, your wagon jolts—stuck. Water rushes past, pulling at the wheels. Panic spreads through your group.",

    image:"images/river.png",

    choices:[

      {text:"Unload supplies to lighten the wagon", flavor:"Lose 10 gold, Progress +1", effects:()=>{gold-=10; progress+=1;}},

      {text:"Force the animals forward", flavor:"50% chance: Lose 2 health OR Progress +2", effects:()=>{if(Math.random()<0.5){health-=2;}else{progress+=2;}}},

      {text:"Secure ropes and try to stabilize the wagon", flavor:"Lose 1 health, Progress +1", effects:()=>{health-=1; progress+=1;}}

    ]

  },

  {

    title:"4. Fever on the Trail",

    text:"It starts with chills…then heat. Your vision blurs. Every step feels heavier. Others keep moving.",

    image:"images/fever.png",

    choices:[

      {text:"Rest and recover", flavor:"Health +2, Progress 0", effects:()=>{health+=2;}},

      {text:"Push forward", flavor:"Lose 3 health, Progress +1", effects:()=>{health-=3; progress+=1;}},

      {text:"Buy medicine from a traveling seller", flavor:"Spend 10 gold, Health +3", effects:()=>{gold-=10; health+=3;}}

    ]

  },

  // --- continue adding all 15 events here similarly ---

];


// DOM ELEMENTS

const introScreen = document.getElementById('intro-screen');

const startBtn = document.getElementById('start-btn');

const eventScreen = document.getElementById('event-screen');

const eventTitle = document.getElementById('event-title');

const eventText = document.getElementById('event-text');

const eventImage = document.getElementById('event-image');

const choicesContainer = document.getElementById('choices-container');

const flavorText = document.getElementById('flavor-text');

const nextBtn = document.getElementById('next-btn');

const healthSpan = document.getElementById('health');

const goldSpan = document.getElementById('gold');

const progressSpan = document.getElementById('progress');


// START BUTTON

startBtn.addEventListener('click', ()=>{

  introScreen.classList.add('hidden');

  eventScreen.classList.remove('hidden');

  showEvent();

});


// SHOW EVENT

function showEvent(){

  const event = events[currentEvent];

  eventTitle.textContent = event.title;

  eventText.textContent = event.text;

  eventImage.src = event.image;

  flavorText.textContent = "";

  choicesContainer.innerHTML = '';

  nextBtn.classList.add('hidden');


  event.choices.forEach((choice, idx)=>{

    const btn = document.createElement('button');

    btn.textContent = choice.text;

    btn.addEventListener('click', ()=>{

      choice.effects();

      flavorText.textContent = choice.flavor;

      updateStats();

      nextBtn.classList.remove('hidden');

      choicesContainer.querySelectorAll('button').forEach(b=>b.disabled=true);

    });

    choicesContainer.appendChild(btn);

  });

}


// NEXT EVENT

nextBtn.addEventListener('click', ()=>{

  currentEvent++;

  if(currentEvent<events.length){

    showEvent();

  } else {

    endGame();

  }

});


function updateStats(){

  healthSpan.textContent = health;

  goldSpan.textContent = gold;

  progressSpan.textContent = progress;

}


function endGame(){

  eventTitle.textContent = "🎉 The Journey Ends";

  eventText.textContent = "You have reached the end of your adventure. Reflect on your story, choices, and results!";

  eventImage.style.display = 'none';

  choicesContainer.innerHTML = '';

  flavorText.innerHTML = `Final Stats: ❤️ ${health}, 💰 ${gold}, 🚶 ${progress}`;

  nextBtn.classList.add('hidden');

}

</script>
